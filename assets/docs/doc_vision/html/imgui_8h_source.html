779 // Our implementation does NOT call c++ constructors because we don't use them in ImGui. Don't use this class as a straight std::vector replacement in your code!
780 template<typename T>
781 class ImVector
782 {
783 public:
784  int Size;
785  int Capacity;
786  T* Data;
787 
788  typedef T value_type;

790  typedef const value_type* const_iterator;

796  inline int size() const { return Size; }
797  inline int capacity() const { return Capacity; }
798 
799  inline value_type& operator[](int i) { IM_ASSERT(i < Size); return Data[i]; }
800  inline const value_type& operator[](int i) const { IM_ASSERT(i < Size); return Data[i]; }
801 
802  inline void clear() { if (Data) { Size = Capacity = 0; ImGui::MemFree(Data); Data = NULL; } }
803  inline iterator begin() { return Data; }
804  inline const_iterator begin() const { return Data; }
805  inline iterator end() { return Data + Size; }
806  inline const_iterator end() const { return Data + Size; }
807  inline value_type& front() { IM_ASSERT(Size > 0); return Data[0]; }
808  inline const value_type& front() const { IM_ASSERT(Size > 0); return Data[0]; }
809  inline value_type& back() { IM_ASSERT(Size > 0); return Data[Size-1]; }
810  inline const value_type& back() const { IM_ASSERT(Size > 0); return Data[Size-1]; }
811  inline void swap(ImVector<T>& rhs) { int rhs_size = rhs.Size; rhs.Size = Size; Size = rhs_size; int rhs_cap = rhs.Capacity; rhs.Capacity = Capacity; Capacity = rhs_cap; value_type* rhs_data = rhs.Data; rhs.Data = Data; Data = rhs_data; }
812 
813  inline int _grow_capacity(int new_size) { int new_capacity = Capacity ? (Capacity + Capacity/2) : 8; return new_capacity > new_size ? new_capacity : new_size; }
814 
815  inline void resize(int new_size) { if (new_size > Capacity) reserve(_grow_capacity(new_size)); Size = new_size; }
816  inline void reserve(int new_capacity)
817  {
818  if (new_capacity <= Capacity) return;
819  T* new_data = (value_type*)ImGui::MemAlloc((size_t)new_capacity * sizeof(value_type));
820  memcpy(new_data, Data, (size_t)Size * sizeof(value_type));
821  ImGui::MemFree(Data);
822  Data = new_data;
823  Capacity = new_capacity;
824  }
825 
826  inline void push_back(const value_type& v) { if (Size == Capacity) reserve(_grow_capacity(Size+1)); Data[Size++] = v; }

828 
829  inline iterator erase(const_iterator it) { IM_ASSERT(it >= begin() && it < end()); const ptrdiff_t off = it - begin(); memmove(Data + off, Data + off + 1, ((size_t)Size - (size_t)off - 1) * sizeof(value_type)); Size--; return Data + off; }
830  inline iterator insert(const_iterator it, const value_type& v) { IM_ASSERT(it >= begin() && it <= end()); const ptrdiff_t off = it - begin(); if (Size == Capacity) reserve(Capacity ? Capacity * 2 : 4); if (off < (int)Size) memmove(Data + off + 1, Data + off, ((size_t)Size - (size_t)off) * sizeof(value_type)); Data[off] = v; Size++; return Data + off; }
